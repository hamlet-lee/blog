# Gradle 创建 Library
```shell
$ mkdir mylib
$ cd mylib

# 这里的mylib会作为artifactId
$ gradle init --type java-library --project-name mylib
```
修改 build.gradle

```gradle
plugins {
    // Apply the java-library plugin to add support for Java Library
    id 'java-library'
}

// 1 加入这两行
group = 'com.xxx.xxx'
version = '1.0.0'

repositories {
    // Use jcenter for resolving your dependencies.
    // You can declare any Maven/Ivy/file repository here.
    jcenter()
}

// 2 加入这一段


task sourcesJar(type: Jar) {
    from sourceSets.main.allJava
    classifier = 'sources'
}

publishing {
    repositories {
        maven {
            // in ~/.gradle/gradle.properties
            // caf_username=xxx
            // caf_password=xxx
            // nexusUrl=http://nexus.xxx.com/nexus/content/repositories/
            // relativePathSnapshots=snapshots/
            // relativePathReleases=releases/

            credentials {
                username caf_username
                password caf_password
            }

             if(project.version.endsWith('-SNAPSHOT')) {
                url = "${nexusUrl}${relativePathSnapshots}"
            } else {
                url = "${nexusUrl}${relativePathReleases}"
            }
        }
    }
    publications {
        mavenJava(MavenPublication) {
            artifact sourcesJar
            groupId = group
            artifactId = project.name
            version = version
            from components.java
        }
    }
}


dependencies {
    // This dependency is exported to consumers, that is to say found on their compile classpath.
    api 'org.apache.commons:commons-math3:3.6.1'

    // This dependency is used internally, and not exposed to consumers on their own compile classpath.
    implementation 'com.google.guava:guava:27.0.1-jre'

    // Use JUnit test framework
    testImplementation 'junit:junit:4.12'
}

// 3 最后调整下编译用的编码格式为UTF-8
tasks.withType(JavaCompile) {  
    options.encoding = "UTF-8"  
}
```
